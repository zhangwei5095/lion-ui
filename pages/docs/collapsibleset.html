说明

多个折叠显示框，当点击其中一个折叠显示框时，此折叠显示框打开，可以设置每次只打开一个，也可以设置打开多个。

前置条件

页面引用JS与CSS文件

<!--整个框架依赖的JS文件-->
<script src="../../src/js/zepto.min.js"></script>
<!--当前组件依赖的JS文件-->
<script src="../../src/js/one.js"></script>
<!--当前组件JS文件-->
<script src="../../src/js/ui/collapsibleset/collapsibleset.js"></script>
<!--当前组件依赖的CSS文件-->
<link href="../../src/css/one.css" rel="stylesheet">
<!--当前组件CSS文件-->
<link href="../../src/css/ui/collapsibleset.css" rel="stylesheet">

高级使用

1.折叠框显示设置

    $(function(){
       //设定第二个折叠框禁止显示
        var collapsibleset=one.ui.collapsibleset('#collapsibleset');
        collapsibleset.allowShowIndex(1,false);
     
       //设定第三个折叠框禁止关闭
        var collapsibleset=one.ui.collapsibleset('#collapsibleset');
        collapsibleset.allowHideIndex(2,false);
     
       //设定全部的折叠框允许显示
         var collapsibleset=one.ui.collapsibleset('#collapsibleset');
         collapsibleset.allowShow(true);
     
       //设定全部的折叠框禁止关闭
         var collapsibleset=one.ui.collapsibleset('#collapsibleset');
         collapsibleset.allowHide(false);
     
       //设定全部的折叠框展开
         var collapsibleset=one.ui.collapsibleset('#collapsibleset');
         collapsibleset.show();
     
       //设定全部的折叠框关闭
         var collapsibleset=one.ui.collapsibleset('#collapsibleset');
         collapsibleset.hide();
     })
