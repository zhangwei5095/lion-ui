前置条件

页面引用JS与CSS文件

<!--整个框架依赖的JS文件-->
<script src="../../src/js/zepto.min.js"></script>
<!--当前组件依赖的JS文件-->
<script src="../../src/js/one.js"></script>
<!--当前组件JS文件-->
<script src="../../src/js/ui/tabs/tabs.js"></script>
<!--当前组件依赖的CSS文件-->
<link href="../../src/css/one.css" rel="stylesheet">
<!--当前组件CSS文件-->
<link href="../../src/css/ui/tabs.css" rel="stylesheet">

高级使用

1.设置切换状态及索引列

    $(function(){
       //初始化组件
        window.tabs = one.ui.tabs(".one_tabs");
     
       //设为不允许切换状态
        tabs.isAllowChoose(false);
     
       //切换至索引三
        tabs.changeTabsTo(3);

     })
